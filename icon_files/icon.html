⚙️ background.js (lógica de tracking)

js
// WebMinutes - MVP
// Cuenta tiempo por URL activa, pausa por inactividad, guarda por día en chrome.storage.local

const IDLE_THRESHOLD_SEC = 45; // si el usuario está inactivo > 45s, no suma tiempo
const SAVE_FLUSH_MS = 5000;    // cada cuánto forzamos guardado (por seguridad)

let current = {
  tabId: null,
  url: null,
  start: null,      // timestamp (ms)
  active: true,     // ventana enfocada + estado activo
};

let lastIdleState = "active";
let lastWindowFocused = true;

// Util: devuelve string YYYY-MM-DD en zona local
function todayKey() {
  const d = new Date();
  const y = d.getFullYear();
  const m = `${d.getMonth() + 1}`.padStart(2, "0");
  const day = `${d.getDate()}`.padStart(2, "0");
  return `${y}-${m}-${day}`;
}

// Guardar acumulado (ms) para una URL en el día actual
async function addTime(url, deltaMs) {
  if (!url || deltaMs <= 0) return;
  const key = todayKey();
  const data = await chrome.storage.local.get([key]);
  const dayMap = data[key] || {}; // { url: ms }
  dayMap[url] = (dayMap[url] || 0) + deltaMs;
  await chrome.storage.local.set({ [key]: dayMap });
}

// Finaliza el conteo de la URL actual si procede
async function finalizeCurrent(reason = "") {
  if (!current.url || current.start == null) return;
  const now = Date.now();
  const elapsed = now - current.start;
  if (elapsed > 0 && current.active) {
    await addTime(current.url, elapsed);
  }
  // Reiniciar marca de inicio
  current.start = now;
}

// Actualiza el target (URL/tab) a trackear
async function setCurrent(tabId, url) {
  await finalizeCurrent("switch");
  current.tabId = tabId;
  current.url = sanitizeUrl(url);
  current.start = Date.now();
}

// Sanea URLs (omitimos chrome://, edge://, about:, etc.)
function sanitizeUrl(url) {
  try {
    if (!url) return null;
    if (url.startsWith("chrome://") || url.startsWith("edge://") || url.startsWith("about:")) return null;
    const u = new URL(url);
    // Opcional: si quisieras trackear por dominio, usarías `${u.origin}`
    // En el MVP pediste granularidad por URL completa:
    return u.href;
  } catch {
    return null;
  }
}

// Detectar pestaña activa inicialmente
async function initActiveTab() {
  const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
  if (tab) {
    await setCurrent(tab.id, tab.url);
  }
}

// Cambios de pestaña activa
chrome.tabs.onActivated.addListener(async (activeInfo) => {
  try {
    const tab = await chrome.tabs.get(activeInfo.tabId);
    await setCurrent(tab.id, tab.url);
  } catch (_) {}
});

// Cambio de URL o carga de una pestaña
chrome.tabs.onUpdated.addListener(async (tabId, changeInfo, tab) => {
  if (tabId === current.tabId && changeInfo.status === "complete") {
    await setCurrent(tabId, tab.url);
  }
  // si cambia la URL antes de "complete", también podríamos actualizar:
  if (tabId === current.tabId && changeInfo.url) {
    await setCurrent(tabId, changeInfo.url);
  }
});

// Enfoque/Desenfoque de ventana
chrome.windows.onFocusChanged.addListener(async (windowId) => {
  const focused = windowId !== chrome.windows.WINDOW_ID_NONE;
  if (focused !== lastWindowFocused) {
    lastWindowFocused = focused;
    current.active = focused && lastIdleState === "active";
    await finalizeCurrent("focus-change");
  }

  if (focused) {
    // Re-identificar pestaña activa al recuperar foco
    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
    if (tab) await setCurrent(tab.id, tab.url);
  }
});

// Estado de actividad del sistema (teclado/ratón)
async function pollIdle() {
  chrome.idle.queryState(IDLE_THRESHOLD_SEC, async (state) => {
    if (state !== lastIdleState) {
      lastIdleState = state;
      current.active = state === "active" && lastWindowFocused;
      await finalizeCurrent("idle-change");
    }
  });
}

// Flush periódico para evitar perder datos si algo se queda colgado
setInterval(async () => {
  await finalizeCurrent("flush");
}, SAVE_FLUSH_MS);

// Inicialización
(async function bootstrap() {
  await initActiveTab();
  // Poll de idle cada ~10s (no necesitamos más frecuencia)
  setInterval(pollIdle, 10000);
})();

js
function todayKey() {
  const d = new Date();
  const y = d.getFullYear();
  const m = `${d.getMonth() + 1}`.padStart(2, "0");
  const day = `${d.getDate()}`.padStart(2, "0");
  return `${y}-${m}-${day}`;
}

function msToHMS(ms) {
  const totalSec = Math.floor(ms / 1000);
  const h = Math.floor(totalSec / 3600);
  const m = Math.floor((totalSec % 3600) / 60);
  const s = totalSec % 60;
  const parts = [];
  if (h > 0) parts.push(`${h}h`);
  if (m > 0) parts.push(`${m}m`);
  parts.push(`${s}s`);
  return parts.join(" ");
}

async function loadToday() {
  const key = todayKey();
  const data = await chrome.storage.local.get([key]);
  const map = data[key] || {}; // { url: ms }

  // Total
  const total = Object.values(map).reduce((a, b) => a + b, 0);
  document.getElementById("totalToday").textContent = msToHMS(total);

  // Top 5
  const entries = Object.entries(map).sort((a, b) => b[1] - a[1]).slice(0, 5);
  const list = document.getElementById("topList");
  list.innerHTML = "";
  entries.forEach(([url, ms]) => {
    const li = document.createElement("li");
    const a = document.createElement("a");
    a.href = url;
    a.textContent = url;
    a.target = "_blank";
    a.rel = "noopener noreferrer";
    a.className = "url";
    const span = document.createElement("span");
    span.className = "time";
    span.textContent = `(${msToHMS(ms)})`;
    li.appendChild(a);
    li.appendChild(span);
    list.appendChild(li);
  });

  // Fecha
  const d = new Date();
  const options = { weekday: "long", day: "2-digit", month: "long" };
  document.getElementById("date").textContent = d.toLocaleDateString(undefined, options);
}

async function resetToday() {
  const key = todayKey();
  await chrome.storage.local.set({ [key]: {} });
  await loadToday();
}

function makeCsvRow(cols) {
  return cols
    .map((c) => `"${String(c).replace(/"/g, '""')}"`)
    .join(",");
}

async function exportCsv() {
  // Exportamos TODO el almacenamiento: varios días
  const all = await chrome.storage.local.get(null);
  // Estructura CSV: date,url,seconds
  let csv = "date,url,seconds\n";
  Object.keys(all).sort().forEach((date) => {
    const dayMap = all[date];
    if (!dayMap || typeof dayMap !== "object") return;
    Object.entries(dayMap).forEach(([url, ms]) => {
      const seconds = Math.floor(ms / 1000);
      csv += makeCsvRow([date, url, seconds]) + "\n";
    });
  });

  const blob = new Blob([csv], { type: "text/csv" });
  const url = URL.createObjectURL(blob);

  // Trigger descarga
  const a = document.createElement("a");
  a.href = url;
  a.download = `webminutes_${Date.now()}.csv`;
  document.body.appendChild(a);
  a.click();
  URL.revokeObjectURL(url);
  a.remove();
}

document.getElementById("btnExport").addEventListener("click", exportCsv);
document.getElementById("btnReset").addEventListener("click", resetToday);
